cmake_minimum_required(VERSION 3.10)
project(ChemicalBalancer)

# 设置C++标准
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# 包含头文件目录
include_directories(include)

# 主程序可执行文件
add_executable(ChemicalBalancer src/main.cpp src/balancer.cpp)

# 测试程序可执行文件
add_executable(test_balancer tests/test_balancer.cpp src/balancer.cpp)
